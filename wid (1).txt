<script>
const REGIONS = {
  'MRM': 'Regiões Montanhosas da Madeira',
  'MCS': 'Costa Sul da Madeira'
};
 
const TYPE_PT = {
  'Agitação Marítima': 'Agitação Marítima',
  'Nevoeiro': 'Nevoeiro',
  'Precipitação': 'Precipitação',
  'Vento': 'Vento',
  'Neve': 'Neve',
  'Trovoada': 'Trovoada',
  'Tempo Frio': 'Tempo Frio',
  'Tempo Quente': 'Tempo Quente',
  'Granizo': 'Granizo',
  'Rain': 'Precipitação',
  'Wind': 'Vento',
  'Fog': 'Nevoeiro',
  'Snow/Ice': 'Neve / Gelo',
  'Thunderstorm': 'Trovoada',
  'Coastal Event': 'Agitação Marítima'
};
 
const LEVEL_ORDER = { red: 4, orange: 3, yellow: 2, green: 1 };
const LEVEL_PT = { green: 'Verde', yellow: 'Amarelo', orange: 'Laranja', red: 'Vermelho' };
 
function maxLevel(warnings) {
  if (!warnings.length) return 'green';
  return warnings.reduce((best, w) => {
    const lv = (w.awarenessLevelID || 'green').toLowerCase();
    return (LEVEL_ORDER[lv] || 0) > (LEVEL_ORDER[best] || 0) ? lv : best;
  }, 'green');
}
 
function formatDatetime(iso) {
  if (!iso) return '';
  try {
    const d = new Date(iso);
    return d.toLocaleDateString('pt-PT', { day: '2-digit', month: '2-digit' })
      + '\n' + d.toLocaleTimeString('pt-PT', { hour: '2-digit', minute: '2-digit' });
  } catch { return iso; }
}
 
function buildRegionCard(code, name, warnings) {
  const active = warnings
    .filter(w => (w.awarenessLevelID || 'green').toLowerCase() !== 'green')
    .sort((a, b) => (LEVEL_ORDER[b.awarenessLevelID] || 0) - (LEVEL_ORDER[a.awarenessLevelID] || 0));
 
  const maxLv = maxLevel(warnings);
 
  let bodyHtml;
  if (active.length === 0) {
    bodyHtml = `
      <div class="all-clear">
        <div class="all-clear-check">✓</div>
        Sem avisos ativos de nível amarelo, laranja ou vermelho
      </div>`;
  } else {
    const rows = active.map(w => {
      const lv = (w.awarenessLevelID || 'green').toLowerCase();
      const typeName = TYPE_PT[w.awarenessTypeName] || w.awarenessTypeName || '—';
      const desc = w.text ? `<div class="w-desc">${w.text}</div>` : '';
      const tStart = formatDatetime(w.startTime);
      const tEnd   = formatDatetime(w.endTime);
      const timeHtml = tStart
        ? `<div>${tStart}</div>${tEnd ? `<div style="color:var(--text-light)">→ ${tEnd}</div>` : ''}`
        : '';
      return `
        <div class="warning-row">
          <div class="w-dot dot-${lv}"></div>
          <div class="w-info">
            <div class="w-type">${typeName}</div>
            ${desc}
          </div>
          <div class="w-time">${timeHtml}</div>
        </div>`;
    }).join('');
    bodyHtml = `<div class="warnings-list">${rows}</div>`;
  }
 
  return `
    <div class="region-card">
      <div class="level-strip strip-${maxLv}"></div>
      <div class="region-header">
        <div class="region-icon" style="background:var(--${maxLv}-dot)"></div>
        <span class="region-name">${name}</span>
        <span class="region-code">${code}</span>
        <span class="level-badge badge-${maxLv}">Nível ${LEVEL_PT[maxLv]}</span>
      </div>
      ${bodyHtml}
    </div>`;
}
 
async function loadWarnings() {
  const content = document.getElementById('content');
  const timeEl  = document.getElementById('update-time');
 
  try {
    const res = await fetch('https://api.ipma.pt/open-data/forecast/warnings/warnings_www.json');
    if (!res.ok) throw new Error('HTTP ' + res.status);
    const data = await res.json();
 
    const now = new Date();
    const regionData = {};
    Object.keys(REGIONS).forEach(k => regionData[k] = []);
 
    data.forEach(w => {
      const area = w.idAreaAviso;
      if (Object.prototype.hasOwnProperty.call(regionData, area)) {
        const endTime = w.endTime ? new Date(w.endTime) : null;
        if (!endTime || endTime > now) {
          regionData[area].push(w);
        }
      }
    });
 
    let html = '';
    Object.entries(REGIONS).forEach(([code, name]) => {
      html += buildRegionCard(code, name, regionData[code]);
    });
 
    content.innerHTML = html;
    timeEl.textContent = 'Atualizado às ' + now.toLocaleTimeString('pt-PT', { hour: '2-digit', minute: '2-digit' });
 
  } catch (err) {
    content.innerHTML = `
      <div class="status-block error">
        <div style="margin-bottom:8px;font-weight:500;">Não foi possível carregar os avisos do IPMA</div>
        <div style="font-size:11px;opacity:0.8;margin-bottom:12px;">${err.message}</div>
        <button class="refresh-btn" onclick="loadWarnings()">↻ Tentar novamente</button>
      </div>`;
    timeEl.textContent = 'Erro na atualização';
  }
}
 
loadWarnings();
setInterval(loadWarnings, 10 * 60 * 1000);
</script>
